SELECT telepulesek.telepNev, COUNT(konyvtarak.id) AS "konyvtarDarab"
FROM telepulesek, konyvtarak
WHERE telepulesek.irsz = konyvtarak.irsz
GROUP BY telepulesek.telepNev
HAVING konyvtarDarab >= 7;
