 SELECT
 telepNev,
 COUNT(id) AS konyvtarDarab
 FROM
 konyvtarak INNER JOIN telepulesek
 ON konyvtarak.irsz=telepulesek.irsz

 telepNev

 konyvtarDarab >=7;
