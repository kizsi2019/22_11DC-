1/
CREATE DATABASE konyvtarak DEFAULT CHARACTER SET utf8
COLLATE utf8_hungarian_ci;
2/
UPDATE megyek SET megyek.megyeNev="Budapest" 
WHERE megyek.megyeNev="BP";
3/
SELECT konyvtarak.konyvtarNev, konyvtarak.irsz FROM konyvtarak
WHERE konyvtarak.konyvtarNev LIKE "%szakkönyvtár";
4/
SELECT konyvtarak.konyvtarNev, konyvtarak.irsz, konyvtarak.cim FROM konyvtarak
WHERE konyvtarak.irsz LIKE "1%" ORDER BY 2 ASC;
5/
SELECT telepulesek.telepNev, COUNT(konyvtarak.id) 
AS "konyvtarDarab" FROM telepulesek, konyvtarak
WHERE telepulesek.irsz = konyvtarak.irsz GROUP BY 1 
HAVING konyvtarDarab>=7; 
6/
SELECT megyek.megyeNev, COUNT(telepulesek.irsz) AS "telepulesDarab"
FROM megyek INNER JOIN telepulesek ON telepulesek.megyeId = megyek.id
WHERE telepulesek.irsz NOT LIKE "1%" GROUP BY 1 ORDER BY 2 DESC;
